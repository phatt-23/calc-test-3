\newcommand*\qs[2]{
	\begin{center}
    \begin{tcolorbox}[
        title=\textbf{#1},
        colback=gre!9!white,
        colframe=gre!40!black,
        colbacklower=gre!40!black,
        skin=bicolor,
        halign lower=flush right,
        collower=white,
		attach boxed title to top left={
			xshift=2mm,
			yshift=-2mm
		},
        boxed title style={
            size=small,
            colback=gre!57!black,
        },
        drop shadow=black!50!white,%
		width=0.7\textwidth,
        bottom=-3mm,
    ]
	    #2
        \tcblower
    \end{tcolorbox}
	\end{center}

}

}

}
